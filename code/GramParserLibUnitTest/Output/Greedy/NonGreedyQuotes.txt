rule doubleSingleQuote = "''";
#	Here we use *? which is the non-greedy version of *
rule quotedString = doubleSingleQuote s:.*? doubleSingleQuote => s;
rule identifier = ("a".."z")+;
rule assignation = id:identifier "=" s:quotedString ";"
	=> { "id":id, "text":s };

rule main = assignation*;
